#!/bin/bash


echo "Запуск числового фильтра..."
echo "-------------------------"

for i in {1..20}; do

        if [ $i -eq 15 ]; then
        echo "-------------------------"
        echo "Достигнуто число 15. Останавливаю цикл."
        break
    fi

    
    if [ $((i % 2)) -eq 0 ]; then
        # Это четное число, пропускаем его и переходим к следующему i
        continue
    fi

       echo "Найдено нечетное число: $i"

done
