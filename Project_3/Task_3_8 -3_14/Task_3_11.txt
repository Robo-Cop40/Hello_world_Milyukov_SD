#!/bin/bash

check_root() {
    if [ "$EUID" -ne 0 ]; then
        echo "ПРЕДУПРЕЖДЕНИЕ: Скрипт должен быть запущен от имени суперпользователя (root)."
        echo "Текущий UID: $EUID. Завершение работы."
        exit 1
    else
        echo "Проверка пройдена: скрипт запущен от имени root (UID=$EUID)."
    fi
}

check_root

echo "Скрипт продолжает работу с правами суперпользователя."
