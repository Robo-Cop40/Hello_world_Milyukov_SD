#!/bin/bash

for i in {1..10}; do
    filename="test${i}.txt"
    touch "$filename"
    echo "  Создан файл: $filename"
done

counter=10


while [ $counter -ge 1 ]; do
    filename="test${counter}.txt"

    # Проверяем, существует ли файл (на всякий случай)
    if [ -f "$filename" ]; then
        rm "$filename"
        echo "  Удален файл: $filename"
    else
        echo "  Файл не найден (странно): $filename"
    fi

       ((counter--))
done
